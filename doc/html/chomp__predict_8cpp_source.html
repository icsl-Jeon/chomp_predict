333 
334  int loop_fps = 20;
335  ros::Rate rate(loop_fps);
336  // in intial phase, prediction will be triggered only if there is a observation
337 
338  bool trigger_prediction = true;
339 
340  double accum_error = 0;
341  double duration_from_last_trigger;
342  ROS_INFO_STREAM("[CHOMP] waiting target observation. We will start if target history is enough(>= No)");
343  while(ros::ok()){
344  if(is_state_received and observation_queue.size() >= pred_param.No){
345  ROS_INFO_ONCE("[CHOMP] enough target observation received. Start loop");
346 
347  // prediction tirgger
348  if(trigger_prediction){
349  predict_with_obsrv_queue();
350  // reset
351  last_prediction_time = ros::Time::now();
352  accum_error = 0;
353  }
354 
355  // check whether target reaches the imminent waypoint. If reached, move to the next one.
356  if (last_obsrv_to_goal() < REACH_TOL and (target_waypoints.size()>1))
357  {to_next_target_waypoint(); is_goal_moved = true;}
358  else
359  is_goal_moved = false;
360 
361 
362  // Check the trigger condition for the next prediction
363  duration_from_last_trigger = (ros::Time::now() - last_prediction_time).toSec();
364  accum_error += 1.0/loop_fps * (pow(eval_prediction(ros::Time::now()).x - observation_queue.back().pose.position.x,2)
365  + pow(eval_prediction(ros::Time::now()).y - observation_queue.back().pose.position.y,2)); // the MSE integration
366 
367  trigger_prediction = (duration_from_last_trigger > pred_param.prediction_horizon - 0.01 ) or
368  (accum_error > pred_param.trigger_tol_accum_error );
369 
370  }
371 
372 
373  // regular routine
374  ros::spinOnce();
375  rate.sleep();
376  // publish routine
377  this->chomp_wrapper.publish_routine(); //inner loop(chomp_ros_wrapper) publisher
378  this->publish_routine(); // outer loop(chomp_predictor) publisher
