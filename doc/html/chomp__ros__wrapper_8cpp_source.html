116 void Wrapper::build_matrix(MatrixXd &M,VectorXd &h,nav_msgs::Path prior_points,geometry_msgs::Point goal,OptimParam* param){
117 
118  // init
119  // if external parameter is given
120  if(param == NULL)
121  optim_param = optim_param_default;
122  else
123  optim_param = *param;
124 
125  // here, A,b is |Ax-b|^2
126  int N = optim_param.n_step;
127  int No = prior_points.poses.size();
128  int m_row = (No + (N-1) + (N-2) + 1) * dim; // prior points + velocity + acceleration + goal
129 
130  MatrixXd A = MatrixXd(m_row,dim*N);
131  VectorXd b = VectorXd(m_row);
132  double gamma = optim_param.gamma;
133 
134 
135  MatrixXd A0 = MatrixXd::Zero(dim*(No+1),dim*N);
136  VectorXd b0 = VectorXd::Zero(dim*(No+1));
137 
138  // 1. prior points + goal fitting
139  for(int n = 0;n<No;n++){
140  double factor = n/N;
141  geometry_msgs::Point p = prior_points.poses[n].pose.position;
142  A0.block(dim*n,dim*n,dim,dim) = (sqrt(exp(gamma*factor)) * VectorXd::Ones(dim)).asDiagonal();
143  b0.block(dim*n,0,dim,1) << sqrt(exp(gamma*factor))*p.x,
144  sqrt(exp(gamma*factor))*p.y;
145  }
146 
147  A0.block(dim*No,dim*(N-1),dim,dim) = (sqrt(exp(gamma*1)) * VectorXd::Ones(dim)).asDiagonal();
148  b0.block(dim*No,0,dim,1) << sqrt(exp(gamma*1))*goal.x,
149  sqrt(exp(gamma*1))*goal.y;
150 
151  // 2. 1st order derivatives
152  MatrixXd A1 = MatrixXd::Zero(dim*(N-1),dim*N);
153  VectorXd b1= VectorXd::Zero(dim*(N-1));
154 
155  for(int n = 0;n<N-1;n++){
156  A1.block((dim*n),(dim*n),dim,dim) = MatrixXd::Identity(dim,dim);
157  A1.block((dim*n),(dim*(n+1)),dim,dim) = -MatrixXd::Identity(dim,dim);
158  }
159 
160  // 3. 2nd order derivatives
161  MatrixXd A2= MatrixXd::Zero(dim*(N-2),dim*N);
162  VectorXd b2= VectorXd::Zero(dim*(N-2));
163 
164  for(int n = 0;n<N-2;n++){
165  A2.block((dim*n),(dim*n),dim,dim) = MatrixXd::Identity(dim,dim);
166  A2.block((dim*n),(dim*(n+1)),dim,dim) = -2*MatrixXd::Identity(dim,dim);
167  A2.block((dim*n),(dim*(n+2)),dim,dim) = MatrixXd::Identity(dim,dim);
168  }
169 
170  // matrix construct
171  A << A0,
172  A1,
173  A2;
174 
175  b << b0,
176  b1,
177  b2;
178 
179 
180  M = MatrixXd(dim*N,dim*N);
181  h = VectorXd(dim*N);
182 
183  M = 2*A.transpose()*A;
184  h = -2*A.transpose()*b;
185 }
