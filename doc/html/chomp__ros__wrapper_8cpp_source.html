113 
114  // init
115  // if external parameter is given
116  if(param == NULL)
117  optim_param = optim_param_default;
118  else
119  optim_param = *param;
120 
121  // here, A,b is |Ax-b|^2
122  int N = optim_param.n_step;
123  int No = prior_points.poses.size();
124  int m_row = (No + (N-1) + (N-2) + 1) * dim; // prior points + velocity + acceleration + goal
125 
126  MatrixXd A = MatrixXd(m_row,dim*N);
127  VectorXd b = VectorXd(m_row);
128  double gamma = optim_param.gamma;
129 
130 
131  MatrixXd A0 = MatrixXd::Zero(dim*(No+1),dim*N);
132  VectorXd b0 = VectorXd::Zero(dim*(No+1));
133 
134  // 1. prior points + goal fitting
135  for(int n = 0;n<No;n++){
136  double factor = n/N;
137  geometry_msgs::Point p = prior_points.poses[n].pose.position;
138  A0.block(dim*n,dim*n,dim,dim) = (sqrt(exp(gamma*factor)) * VectorXd::Ones(dim)).asDiagonal();
139  b0.block(dim*n,0,dim,1) << sqrt(exp(gamma*factor))*p.x,
140  sqrt(exp(gamma*factor))*p.y;
141  }
142 
143  A0.block(dim*No,dim*(N-1),dim,dim) = (sqrt(exp(gamma*1)) * VectorXd::Ones(dim)).asDiagonal();
144  b0.block(dim*No,0,dim,1) << sqrt(exp(gamma*1))*goal.x,
145  sqrt(exp(gamma*1))*goal.y;
146 
147  // 2. 1st order derivatives
148  MatrixXd A1 = MatrixXd::Zero(dim*(N-1),dim*N);
149  VectorXd b1= VectorXd::Zero(dim*(N-1));
150 
151  for(int n = 0;n<N-1;n++){
152  A1.block((dim*n),(dim*n),dim,dim) = MatrixXd::Identity(dim,dim);
153  A1.block((dim*n),(dim*(n+1)),dim,dim) = -MatrixXd::Identity(dim,dim);
154  }
155 
156  // 3. 2nd order derivatives
157  MatrixXd A2= MatrixXd::Zero(dim*(N-2),dim*N);
158  VectorXd b2= VectorXd::Zero(dim*(N-2));
159 
160  for(int n = 0;n<N-2;n++){
161  A2.block((dim*n),(dim*n),dim,dim) = MatrixXd::Identity(dim,dim);
162  A2.block((dim*n),(dim*(n+1)),dim,dim) = -2*MatrixXd::Identity(dim,dim);
163  A2.block((dim*n),(dim*(n+2)),dim,dim) = MatrixXd::Identity(dim,dim);
164  }
165 
166  // matrix construct
167  A << A0,
168  A1,
169  A2;
170 
171  b << b0,
172  b1,
173  b2;
174 
175 
176  M = MatrixXd(dim*N,dim*N);
177  h = VectorXd(dim*N);
178 
179  M = 2*A.transpose()*A;
180  h = -2*A.transpose()*b;
181 }
